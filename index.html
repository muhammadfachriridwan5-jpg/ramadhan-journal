import React, { useState, useEffect } from 'react';

import { 

  Moon, Sun, BookOpen, CheckCircle, PenTool, 

  Calendar, Printer, Heart, Star, ChevronRight, 

  ChevronLeft, Droplets, Utensils, Zap, Sparkles, 

  Gift, List, User, MapPin, Phone, Coffee

} from 'lucide-react';


export default function RamadhanPlanner() {

  const [activeTab, setActiveTab] = useState('cover');

  const [currentDay, setCurrentDay] = useState(1);

  const [showPrintMode, setShowPrintMode] = useState(false);

  const [animating, setAnimating] = useState(false);

  const [randomQuote, setRandomQuote] = useState('');

  

  // Database Quotes & Motivasi

  const quotes = [

    "Lelahmu saat ibadah akan hilang, tapi pahalanya akan kekal.",

    "Ramadhan itu bukan hanya menahan lapar, tapi menahan ego.",

    "Jangan sampai puasa cuma dapat lapar dan haus. Yuk, jaga hati.",

    "Allah melihat usahamu, bukan kesempurnaanmu. Teruslah berjuang.",

    "Sujud itu indah, kamu berbisik ke bumi tapi didengar di langit.",

    "Mungkin ini Ramadhan terakhir kita. Lakukan yang terbaik.",

    "Jarak terdekat masalahmu dengan solusinya adalah jarak keningmu dengan sajadah.",

    "Dunia ini cuma sebentar, akhirat selamanya. Prioritaskan yang abadi.",

    "Al-Quran yang berdebu adalah cerminan hati yang mulai kaku.",

    "Tuhan tidak menuntutmu berlari, Dia hanya memintamu untuk tidak berhenti melangkah kepada-Nya."

  ];


  // State untuk menyimpan data user

  const [userData, setUserData] = useState({

    // Profil (Page 2)

    name: '',

    ttl: '',

    address: '',

    phone: '',

    motto: '',

    goals: '',

    

    // Intro (Page 5 & 13)

    listDoa: '',

    kebiasaanBuruk: '',

    kebiasaanBaik: '',

    komitmen: '',


    // Target (Page 15)

    quranTarget: '1', 

    

    // Tracker (Page 12)

    utangPuasa: [], // Array tanggal yang bolong


    // Daily Logs (Page 17-52)

    dailyLogs: {},

    

    // Checklist Sunnah Umum

    checklist: {

      tarawih: false,

      sedekah: false,

      itikaf: false,

      tahajud: false,

      duha: false

    },

    

    // Refleksi (Page 28, 34, 40)

    reflections: {

      start: '',

      mid: '',

      end: ''

    },


    // Idul Fitri & Syawal (Page 55 & 58)

    eidSunnah: {

      mandi: false,

      pakaian: false,

      makan: false,

      takbir: false,

      tahniah: false,

      jalan: false

    },

    syawalFasting: [false, false, false, false, false, false]

  });


  useEffect(() => {

    const saved = localStorage.getItem('ramadhanPlannerComplete');

    if (saved) {

      setUserData(JSON.parse(saved));

    }

    shuffleQuote();

  }, []);


  useEffect(() => {

    localStorage.setItem('ramadhanPlannerComplete', JSON.stringify(userData));

  }, [userData]);


  const shuffleQuote = () => {

    const random = quotes[Math.floor(Math.random() * quotes.length)];

    setRandomQuote(random);

  };


  const changeTab = (tab) => {

    if (tab === activeTab) return;

    setAnimating(true);

    setTimeout(() => {

      setActiveTab(tab);

      setAnimating(false);

    }, 300);

  };


  const updateDailyLog = (day, field, value) => {

    setUserData(prev => ({

      ...prev,

      dailyLogs: {

        ...prev.dailyLogs,

        [day]: {

          ...prev.dailyLogs[day],

          [field]: value

        }

      }

    }));

  };


  const toggleUtangPuasa = (day) => {

    setUserData(prev => {

      const current = prev.utangPuasa || [];

      if (current.includes(day)) {

        return { ...prev, utangPuasa: current.filter(d => d !== day) };

      } else {

        return { ...prev, utangPuasa: [...current, day] };

      }

    });

  };


  const handlePrint = () => {

    setShowPrintMode(true);

    setTimeout(() => {

      window.print();

      setShowPrintMode(false);

    }, 100);

  };


  // --- Components ---


  const CoverSection = () => (

    <div className="h-full flex flex-col space-y-6 overflow-y-auto custom-scrollbar pr-2">

      {/* Cover Header */}

      <div className="flex-shrink-0 flex flex-col items-center justify-center text-center p-6 sm:p-8 border-4 border-double border-blue-200/30 rounded-xl bg-gradient-to-br from-blue-900 via-blue-800 to-slate-900 text-white shadow-inner relative overflow-hidden group min-h-[40%]">

         <div className="absolute top-0 left-0 w-full h-full opacity-10 pointer-events-none">

            <div className="absolute top-10 left-10 w-32 h-32 bg-yellow-400 rounded-full blur-3xl"></div>

         </div>

         <Moon size={50} className="text-yellow-300 mb-4 drop-shadow-[0_0_15px_rgba(253,224,71,0.5)]" />

         <h1 className="text-4xl sm:text-6xl font-serif font-bold tracking-wider mb-2">RAMADHAN</h1>

         <h2 className="text-xl sm:text-2xl font-light tracking-[0.4em] text-blue-200 uppercase">Journal</h2>

         <p className="mt-4 text-sm sm:text-lg text-blue-300 font-mono">1447 H / 2026 M</p>

      </div>


      {/* About Me Section (Page 2 PDF) */}

      <div className="bg-white p-6 rounded-xl shadow-lg border border-slate-100 flex-grow">

         <h3 className="text-xl font-bold text-blue-900 mb-6 flex items-center gap-2 border-b pb-2">

            <User size={20} /> About Me

         </h3>

         <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">

            <div className="space-y-1">

               <label className="text-[10px] uppercase font-bold text-slate-400">Nama Lengkap</label>

               <input type="text" className="w-full border-b border-blue-200 py-1 focus:outline-none focus:border-blue-500 text-blue-900" 

                  value={userData.name} onChange={(e) => setUserData({...userData, name: e.target.value})} placeholder="..." />

            </div>

            <div className="space-y-1">

               <label className="text-[10px] uppercase font-bold text-slate-400">TTL</label>

               <input type="text" className="w-full border-b border-blue-200 py-1 focus:outline-none focus:border-blue-500 text-blue-900" 

                  value={userData.ttl} onChange={(e) => setUserData({...userData, ttl: e.target.value})} placeholder="..." />

            </div>

            <div className="space-y-1 sm:col-span-2">

               <label className="text-[10px] uppercase font-bold text-slate-400 flex items-center gap-1"><MapPin size={10}/> Alamat</label>

               <input type="text" className="w-full border-b border-blue-200 py-1 focus:outline-none focus:border-blue-500 text-blue-900" 

                  value={userData.address} onChange={(e) => setUserData({...userData, address: e.target.value})} placeholder="..." />

            </div>

            <div className="space-y-1">

               <label className="text-[10px] uppercase font-bold text-slate-400 flex items-center gap-1"><Phone size={10}/> Phone</label>

               <input type="text" className="w-full border-b border-blue-200 py-1 focus:outline-none focus:border-blue-500 text-blue-900" 

                  value={userData.phone} onChange={(e) => setUserData({...userData, phone: e.target.value})} placeholder="..." />

            </div>

             <div className="space-y-1">

               <label className="text-[10px] uppercase font-bold text-slate-400">Motto</label>

               <input type="text" className="w-full border-b border-blue-200 py-1 focus:outline-none focus:border-blue-500 text-blue-900" 

                  value={userData.motto} onChange={(e) => setUserData({...userData, motto: e.target.value})} placeholder="..." />

            </div>

         </div>

      </div>

    </div>

  );


  const IntroSection = () => (

    <div className="h-full flex flex-col space-y-6 overflow-y-auto pr-2 custom-scrollbar">

      {/* Intro Narrative (Page 4 PDF) */}

      <div className="bg-blue-50 rounded-xl p-6 border-l-4 border-blue-600">

        <h3 className="text-xl font-bold text-blue-900 mb-2">Sebenarnya siapa yang kita cari?</h3>

        <p className="text-sm text-slate-600 leading-relaxed text-justify">

          Hai, kamu yang sedang memegang planner ini. Kita sama, kan? Sama-sama manusia yang penuh retak dan luka. 

          Aku tidak mau Ramadanmu tahun ini cuma lewat jadi rutinitas 'pindah jam makan'. Aku mau kita beneran pulang. 

          Pulang ke diri kita yang asli, pulang ke pelukan Tuhan.

        </p>

      </div>


      {/* Komitmen (Page 13 PDF) */}

      <div className="bg-white p-6 rounded-xl shadow-md border border-slate-100 space-y-4">

         <h3 className="font-bold text-blue-900 flex items-center gap-2"><Heart size={18} className="text-pink-500"/> Komitmen Ramadhan</h3>

         

         <div>

            <label className="text-xs font-bold text-slate-500 block mb-1">Kebiasaan yang ingin dikurangi?</label>

            <input type="text" className="w-full bg-slate-50 p-2 rounded text-sm border border-slate-200 focus:border-blue-400 focus:outline-none"

               value={userData.kebiasaanBuruk} onChange={(e) => setUserData({...userData, kebiasaanBuruk: e.target.value})} placeholder="Contoh: Ghibah, Marah-marah..." />

         </div>

         <div>

            <label className="text-xs font-bold text-slate-500 block mb-1">Hal yang ingin dijaga?</label>

            <input type="text" className="w-full bg-slate-50 p-2 rounded text-sm border border-slate-200 focus:border-blue-400 focus:outline-none"

               value={userData.kebiasaanBaik} onChange={(e) => setUserData({...userData, kebiasaanBaik: e.target.value})} placeholder="Contoh: Shalat tepat waktu..." />

         </div>

      </div>


      {/* List Doa (Page 5 PDF) */}

      <div className="bg-gradient-to-br from-indigo-600 to-blue-800 p-6 rounded-xl shadow-xl text-white flex-grow">

         <h3 className="font-bold text-lg mb-1 flex items-center gap-2"><List size={18} className="text-yellow-300"/> List Doa Mustajab</h3>

         <p className="text-xs text-blue-200 mb-4 italic">Langitkan di waktu mustajab (Saat berbuka, Sahur, Sujud).</p>

         <textarea 

            className="w-full h-full min-h-[150px] bg-white/10 rounded-lg border border-white/20 p-4 text-sm text-white placeholder-blue-300 focus:outline-none focus:bg-white/20 transition-colors resize-none"

            placeholder="Tuliskan semua hajat dan impianmu di sini..."

            value={userData.listDoa}

            onChange={(e) => setUserData({...userData, listDoa: e.target.value})}

         />

      </div>

    </div>

  );


  const TargetSection = () => {

    const pagesPerJuz = 20;

    const targetJuz = parseInt(userData.quranTarget) || 1;

    const pagesPerDay = targetJuz * pagesPerJuz;

    

    // Utang Puasa count

    const missedDays = userData.utangPuasa?.length || 0;


    return (

      <div className="h-full flex flex-col space-y-6 overflow-y-auto pr-2 custom-scrollbar">

        {/* Quran Tracker (Page 15 PDF) */}

        <div className="bg-white rounded-xl shadow-lg overflow-hidden border border-slate-100 flex-shrink-0">

          <div className="bg-blue-900 p-6 text-white text-center relative">

            <BookOpen size={32} className="mx-auto mb-2 text-blue-300" />

            <h2 className="text-2xl font-bold">Quran Tracker</h2>

            <div className="flex justify-center mt-4 bg-blue-800/50 inline-flex rounded-lg p-1">

               {[0.5, 1, 2, 3].map(val => (

                 <button key={val} onClick={() => setUserData({...userData, quranTarget: val})}

                   className={`px-3 py-1 rounded text-xs font-bold transition-all ${parseFloat(userData.quranTarget) === val ? 'bg-yellow-400 text-blue-900' : 'text-blue-200 hover:text-white'}`}>

                   {val} Juz

                 </button>

               ))}

            </div>

            <p className="mt-2 text-xs text-blue-200">Target: {pagesPerDay} Halaman / Hari</p>

          </div>

          <div className="p-4 grid grid-cols-6 sm:grid-cols-10 gap-2">

             {[...Array(30)].map((_, i) => (

                <button key={i} onClick={() => {

                    const k = `juz_${i+1}`; 

                    setUserData(p => ({...p, checklist: {...p.checklist, [k]: !p.checklist[k]}}));

                }} 

                className={`aspect-square rounded flex items-center justify-center text-xs font-bold border transition-colors ${userData.checklist[`juz_${i+1}`] ? 'bg-blue-600 text-white border-blue-600' : 'bg-slate-50 text-slate-400 border-slate-100'}`}>

                   {i+1}

                </button>

             ))}

          </div>

        </div>


        {/* Utang Puasa (Page 12 PDF) */}

        <div className="bg-rose-50 rounded-xl p-6 border border-rose-100 shadow-sm flex-grow">

           <div className="flex justify-between items-start mb-4">

              <div>

                 <h3 className="text-xl font-bold text-rose-900 flex items-center gap-2"><Coffee size={20}/> Utang Puasa</h3>

                 <p className="text-xs text-rose-700 mt-1">Tandai hari jika kamu berhalangan/tidak puasa.</p>

              </div>

              <div className="bg-white px-3 py-1 rounded-lg border border-rose-200 shadow-sm">

                 <span className="text-2xl font-bold text-rose-600">{missedDays}</span> <span className="text-xs text-rose-400 font-bold">HARI</span>

              </div>

           </div>

           

           <div className="grid grid-cols-7 gap-2">

              {[...Array(30)].map((_, i) => {

                 const day = i + 1;

                 const isMissed = (userData.utangPuasa || []).includes(day);

                 return (

                    <button key={i} onClick={() => toggleUtangPuasa(day)}

                       className={`aspect-square rounded-lg flex items-center justify-center font-bold text-sm transition-all ${isMissed ? 'bg-rose-500 text-white shadow-inner' : 'bg-white text-slate-300 border border-slate-200 hover:border-rose-300'}`}>

                       {day}

                    </button>

                 )

              })}

           </div>

           <p className="text-[10px] text-rose-500/70 mt-4 italic text-center">*Ingat untuk menggantinya (Qadha) setelah Ramadhan.</p>

        </div>

      </div>

    );

  };


  const DailyPlanner = () => {

    const data = userData.dailyLogs[currentDay] || {};

    const prayers = ['Subuh', 'Dzuhur', 'Ashar', 'Maghrib', 'Isya'];


    return (

      <div className="h-full flex flex-col space-y-4">

        {/* Quote Banner */}

        <div className="bg-gradient-to-r from-indigo-500 via-purple-500 to-pink-500 rounded-xl p-3 sm:p-4 text-white shadow-md flex justify-between items-center gap-4 shrink-0">

           <div className="flex items-start gap-2 sm:gap-3">

             <Sparkles size={18} className="mt-1 shrink-0 text-yellow-300" />

             <p className="text-xs sm:text-sm font-medium leading-tight italic">"{randomQuote}"</p>

           </div>

           <button onClick={shuffleQuote} className="shrink-0 p-2 bg-white/20 hover:bg-white/30 rounded-full transition-colors" title="Quote Baru">

             <Zap size={14} className="text-yellow-200" />

           </button>

        </div>


        {/* Day Navigation */}

        <div className="flex items-center justify-between bg-white p-2 sm:p-3 rounded-xl shadow-sm border border-slate-100 shrink-0 sticky top-0 z-10">

          <button onClick={() => setCurrentDay(c => Math.max(1, c - 1))} disabled={currentDay === 1} className="p-2 hover:bg-slate-50 rounded-lg text-blue-600 disabled:opacity-30"><ChevronLeft size={20} /></button>

          <div className="text-center">

            <h2 className="text-lg sm:text-xl font-bold text-blue-900">Ramadhan Day {currentDay}</h2>

            <div className="flex gap-1 justify-center mt-1">

               {[...Array(3)].map((_,i) => <div key={i} className="w-1 h-1 rounded-full bg-blue-300"></div>)}

            </div>

          </div>

          <button onClick={() => setCurrentDay(c => Math.min(30, c + 1))} disabled={currentDay === 30} className="p-2 hover:bg-slate-50 rounded-lg text-blue-600 disabled:opacity-30"><ChevronRight size={20} /></button>

        </div>


        {/* Scrollable Content */}

        <div className="flex-grow overflow-y-auto custom-scrollbar pr-1 space-y-4 pb-20 sm:pb-4">

           {/* Shalat Tracker */}

           <div className="bg-white rounded-xl shadow-sm border border-slate-100 overflow-hidden">

              <div className="bg-slate-50 px-4 py-2 border-b border-slate-100 flex justify-between items-center">

                 <h3 className="font-bold text-blue-800 text-xs sm:text-sm">Ibadah Wajib</h3>

                 <div className="flex gap-2 text-[8px] uppercase font-bold text-slate-400"><span>Qob</span><span>Wajib</span><span>Bad</span></div>

              </div>

              <div className="divide-y divide-slate-50">

                {prayers.map((prayer) => (

                   <div key={prayer} className="grid grid-cols-[1fr_auto_auto_auto] gap-3 px-4 py-2 items-center">

                      <span className="font-semibold text-slate-700 text-sm">{prayer}</span>

                      <input type="checkbox" className="w-4 h-4 accent-blue-600" checked={data[`${prayer}_qobliyah`] || false} onChange={(e) => updateDailyLog(currentDay, `${prayer}_qobliyah`, e.target.checked)}/>

                      <button onClick={() => updateDailyLog(currentDay, `${prayer}_wajib`, !data[`${prayer}_wajib`])} className={`w-12 py-1 rounded text-[10px] font-bold ${data[`${prayer}_wajib`] ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-400'}`}>{data[`${prayer}_wajib`] ? 'OK' : '-'}</button>

                      {['Subuh', 'Ashar'].includes(prayer) ? <div className="w-4 text-center">-</div> : <input type="checkbox" className="w-4 h-4 accent-blue-600" checked={data[`${prayer}_badiyah`] || false} onChange={(e) => updateDailyLog(currentDay, `${prayer}_badiyah`, e.target.checked)}/>}

                   </div>

                ))}

              </div>

           </div>


           <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">

              {/* Sunnah */}

              <div className="bg-blue-600 rounded-xl shadow p-4 text-white">

                 <h3 className="font-bold mb-3 flex items-center gap-2 text-sm border-b border-white/20 pb-2"><Star size={14} className="text-yellow-300 fill-yellow-300"/> SUNNAH</h3>

                 {['Tahajud', 'Dhuha', 'Tilawah', 'Sedekah'].map(item => (

                    <label key={item} className="flex items-center gap-3 cursor-pointer py-1">

                       <div className={`w-4 h-4 rounded border flex items-center justify-center ${data[`sunnah_${item}`] ? 'bg-white text-blue-600' : 'border-white/50'}`}>{data[`sunnah_${item}`] && <CheckCircle size={10} />}</div>

                       <input type="checkbox" className="hidden" checked={data[`sunnah_${item}`] || false} onChange={(e) => updateDailyLog(currentDay, `sunnah_${item}`, e.target.checked)}/>

                       <span className="text-sm">{item}</span>

                    </label>

                 ))}

              </div>

              

              {/* Makan Minum */}

              <div className="bg-white rounded-xl shadow border border-slate-100 p-4">

                 <h3 className="font-bold text-slate-700 mb-3 text-sm flex items-center gap-2"><Utensils size={14}/> NUTRISI</h3>

                 <div className="space-y-2">

                    <input type="text" className="w-full bg-slate-50 border-none rounded px-2 py-1 text-sm placeholder-slate-400" value={data.sahur || ''} onChange={(e) => updateDailyLog(currentDay, 'sahur', e.target.value)} placeholder="Menu Sahur"/>

                    <input type="text" className="w-full bg-slate-50 border-none rounded px-2 py-1 text-sm placeholder-slate-400" value={data.iftar || ''} onChange={(e) => updateDailyLog(currentDay, 'iftar', e.target.value)} placeholder="Menu Iftar"/>

                    <div className="flex gap-1 pt-1 justify-between">

                       {[1,2,3,4,5,6,7,8].map(g => (

                          <button key={g} onClick={() => updateDailyLog(currentDay, 'water', g)} className={`h-6 flex-1 rounded-sm ${data.water >= g ? 'bg-blue-400' : 'bg-slate-100'}`}/>

                       ))}

                    </div>

                 </div>

              </div>

           </div>

           

           {/* Catatan */}
